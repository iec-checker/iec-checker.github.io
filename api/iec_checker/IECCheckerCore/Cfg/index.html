type bb_ty =
  1. | BB
    (*

    Regular basic block

    *)
  2. | BBEntry
    (*

    Point of entry

    *)
  3. | BBExit
    (*

    Point of exit

    *)
  4. | BBJump
    (*

    Indirect jump to/from a node

    *)

Type of a basic block.

type bb = {
  1. id : int;
  2. mutable ty : bb_ty;
  3. mutable preds : int list;
    (*

    Ids of predecessor nodes

    *)
  4. mutable succs : int list;
    (*

    Ids of successor nodes

    *)
  5. mutable stmts : S.statement list;
    (*

    Statements that makes up this BB

    *)
}
